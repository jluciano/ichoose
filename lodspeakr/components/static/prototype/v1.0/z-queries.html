Q1. Is certification standard openly published--?

PREFIX certin:
PREFIX rdfs:
SELECT DISTINCT ?Standard ?OpenlyPublishedAt
WHERE
{ ?x rdfs:label ?Standard .
?x certin:standardURI ?OpenlyPublishedAt .
FILTER regex(?Standard, "FLO")}

Q2. Are certification compliance criteria/control points openly published--?


PREFIX certin:
PREFIX rdfs:
SELECT DISTINCT ?StandardBody ?Certifier ?ComplianceCriteria_PublishedAt
WHERE
{ ?x rdfs:label ?StandardBody .
?x rdfs:label ?Certifier .
?x certin:complianceCriteriaURI ?ComplianceCriteria_PublishedAt .
FILTER regex(?Certifier, "Fairtrade USA")
}

Q4. Can know date of the inspection


prefix dcterms:
prefix rdfs:
prefix certin:

SELECT DISTINCT ?FLOID ?Inspection ?Date ?Inspector ?Certifier
WHERE {
?x rdfs:label ?Inspection .
?x certin:hasInspector ?i .
?i rdfs:label ?Inspector .
?x dcterms:date ?Date .
?x certin:hasCertifier ?c .
?c rdfs:label ?Certifier .
?x certin:hasApplicant ?f .
?f dcterms:identifier ?FLOID .
FILTER regex(?FLOID, "2178")}
ORDER BY ?Inspection

Q7. Can know who is the inspector/auditor


prefix dcterms:
prefix rdfs:
prefix certin:

SELECT DISTINCT ?FLOID ?Inspection ?Inspector ?Inspector_License
WHERE {
?x rdfs:label ?Inspection .
?x certin:hasInspector ?i .
?i rdfs:label ?Inspector .
?x certin:hasApplicant ?f .
?f dcterms:identifier ?FLOID .
FILTER regex(?FLOID, "2178")
OPTIONAL {
?x dcterms:license ?y .
?y rdfs:label ?Inspector_License }
}
ORDER BY ?Inspection
